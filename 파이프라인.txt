pipeline {
 agent any
 
 environment {
   GIT_URL = "https://github.com/siwon-park/temp_repo.git"
 }
 
 stages {
   stage('Pull') {
     steps {
       script {
         git url: "${GIT_URL}", branch: "master", credentialsId: "jenkins_token", poll: true, changelog: true
       }
     }
   }
   
   stage('SpringBoot Build') {
     steps {
       script {
         dir('testapp') {
             sh 'chmod +x gradlew'
             sh './gradlew build'
         }
       }
     }
   }
   
   stage('Build') {
     steps {
       script {
           dir('testapp'){
               sh 'docker build -t springboot .'
         }
       }
     }
   }
   
   stage('Deploy') {
     steps {
       script {
         sh 'docker stop springboot && docker rm springboot'
         sh 'docker run -d --name springboot -p 8080:8080'
       }
     }
   }
   
   stage('Finish') {
     steps {
       script {
         sh 'docker images -qf dangling=true | xargs -I{} docker rmi {}'
       }
     }
   }
 }
}
